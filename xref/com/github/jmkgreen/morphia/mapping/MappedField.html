88       * Discovers interesting (that we care about) things about the field.
89       */
90      protected void discover() {
91          for (Class<? extends Annotation> clazz : interestingAnnotations)
92              addAnnotation(clazz);
93  
94          //type must be discovered before the constructor.
95          realType = discoverType();
96          ctor = discoverCTor();
97          discoverMultivalued();
98  
99          // check the main type
100         isMongoType = ReflectionUtils.isPropertyType(realType);
101 
102         // if the main type isn't supported by the Mongo, see if the subtype is.
103         // works for T[], List<T>, Map<?, T>, where T is Long/String/etc.
104         if (!isMongoType && subType != null)
105             isMongoType = ReflectionUtils.isPropertyType(subType);
106 
107         if (!isMongoType && !isSingleValue && (subType == null || subType.equals(Object.class))) {
108             if (log.isWarningEnabled())
109                 log.warning("The multi-valued field '"
110                         + getFullName()
111                         + "' is a possible heterogenous collection. It cannot be verified. Please declare a valid type to get rid of this warning. " + subType);
112             isMongoType = true;
113         }
114     }

116     private void discoverMultivalued() {
117         if (realType.isArray() ||
118                 Collection.class.isAssignableFrom(realType) ||
119                 Map.class.isAssignableFrom(realType)) {
120 
121             isSingleValue = false;
122 
123             isMap = Map.class.isAssignableFrom(realType);
124             isSet = Set.class.isAssignableFrom(realType);
125             //for debugging
126             isCollection = Collection.class.isAssignableFrom(realType);
127             isArray = realType.isArray();
128 
129             //for debugging with issue
130             if (!isMap && !isSet && !isCollection && !isArray)
131                 throw new MappingException("type is not a map/set/collection/array : " + realType);
132 
133             // get the subtype T, T[]/List<T>/Map<?,T>; subtype of Long[], List<Long> is Long
134             subType = (realType.isArray()) ? realType.getComponentType() : ReflectionUtils.getParameterizedType(field, (isMap) ? 1 : 0);
135 
136             if (isMap)
137                 mapKeyType = ReflectionUtils.getParameterizedType(field, 0);
138         }
139     }

141     private Class discoverType() {
142         Class type = field.getType();
143         Type gType = field.getGenericType();
144         TypeVariable<GenericDeclaration> tv = null;
145         ParameterizedType pt = null;
146         if (gType instanceof TypeVariable)
147             tv = (TypeVariable<GenericDeclaration>) gType;
148         else if (gType instanceof ParameterizedType)
149             pt = (ParameterizedType) gType;
150 
151         if (tv != null) {
152 //			type = ReflectionUtils.getTypeArgument(persistedClass, tv);
153             Class typeArgument = ReflectionUtils.getTypeArgument(persistedClass, tv);
154             if (typeArgument != null)
155                 type = typeArgument;
156         } else if (pt != null) {
157             if (log.isDebugEnabled())
158                 log.debug("found instance of ParameterizedType : " + pt);
159         }
160 
161         if (Object.class.equals(realType) && (tv != null || pt != null))
162             if (log.isWarningEnabled())
163                 log.warning("Parameterized types are treated as untyped Objects. See field '" + field.getName() + "' on " + field.getDeclaringClass());
164 
165         if (type == null)
166             throw new MappingException("A type could not be found for " + this.field);
167 
168         return type;
169     }

171     private Constructor discoverCTor() {
172         Constructor returnCtor = null;
173         Class ctorType = null;
174         // get the first annotation with a concreteClass that isn't Object.class
175         for (Annotation an : foundAnnotations.values()) {
176             try {
177                 Method m = an.getClass().getMethod("concreteClass");
178                 m.setAccessible(true);
179                 Object o = m.invoke(an);
180                 if (o != null && !(o.equals(Object.class))) {
181                     ctorType = (Class) o;
182                     break;
183                 }
184             } catch (NoSuchMethodException e) {
185                 // do nothing
186             } catch (IllegalArgumentException e) {
187                 if (log.isWarningEnabled())
188                     log.warning("There should not be an argument", e);
189             } catch (Exception e) {
190                 if (log.isWarningEnabled())
191                     log.warning("", e);
192             }
193         }
194 
195         if (ctorType != null)
196             try {
197                 returnCtor = ctorType.getDeclaredConstructor();
198                 returnCtor.setAccessible(true);
199             } catch (NoSuchMethodException e) {
200                 if (!hasAnnotation(ConstructorArgs.class))
201                     if (log.isWarningEnabled())
202                         log.warning("No usable constructor for " + ctorType.getName(), e);
203             }
204         else {
205             // see if we can create instances of the type used for declaration
206             ctorType = getType();
207             if (ctorType != null)
208                 try {
209                     returnCtor = ctorType.getDeclaredConstructor();
210                     returnCtor.setAccessible(true);
211                 } catch (NoSuchMethodException e) {
212                     // never mind.
213                 } catch (SecurityException e) {
214                     // never mind.
215                 }
216         }
217         return returnCtor;
218     }

245         String fieldName = getNameToStore();
246         boolean foundField = false;
247         for (String n : getLoadNames()) {
248             if (dbObj.containsField(n))
249                 if (!foundField) {
250                     foundField = true;
251                     fieldName = n;

253                     throw new MappingException(String.format("Found more than one field from @AlsoLoad %s", getLoadNames()));

319     /**
320      * Returns the name of the field's key-name for mongodb
321      */
322     private String getMappedFieldName() {
323         if (hasAnnotation(Id.class))
324             return Mapper.ID_KEY;
325         else if (hasAnnotation(Property.class)) {
326             Property mv = (Property) foundAnnotations.get(Property.class);
327             if (!mv.value().equals(Mapper.IGNORED_FIELDNAME))
328                 return mv.value();
329         } else if (hasAnnotation(Reference.class)) {
330             Reference mr = (Reference) foundAnnotations.get(Reference.class);
331             if (!mr.value().equals(Mapper.IGNORED_FIELDNAME))
332                 return mr.value();
333         } else if (hasAnnotation(Embedded.class)) {
334             Embedded me = (Embedded) foundAnnotations.get(Embedded.class);
335             if (!me.value().equals(Mapper.IGNORED_FIELDNAME))
336                 return me.value();
337         } else if (hasAnnotation(Serialized.class)) {
338             Serialized me = (Serialized) foundAnnotations.get(Serialized.class);
339             if (!me.value().equals(Mapper.IGNORED_FIELDNAME))
340                 return me.value();
341         } else if (hasAnnotation(Version.class)) {
342             Version me = (Version) foundAnnotations.get(Version.class);
343             if (!me.value().equals(Mapper.IGNORED_FIELDNAME))
344                 return me.value();
345         }
346 
347         return this.field.getName();
348     }

350     @Override
351     public String toString() {
352         StringBuffer sb = new StringBuffer();
353         sb.append(getMappedFieldName()).append(" (");
354         sb.append(" type:").append(realType.getSimpleName()).append(",");
355 
356         if (isSingleValue())
357             sb.append(" single:true,");
358         else {
359             sb.append(" multiple:true,");
360             sb.append(" subtype:").append(getSubClass()).append(",");
361         }
362         if (isMap()) {
363             sb.append(" map:true,");
364             if (getMapKeyClass() != null)
365                 sb.append(" map-key:").append(getMapKeyClass().getSimpleName());
366             else
367                 sb.append(" map-key: class unknown! ");
368         }
369 
370         if (isSet())
371             sb.append(" set:true,");
372         if (isCollection)
373             sb.append(" collection:true,");
374         if (isArray)
375             sb.append(" array:true,");
376 
377         //remove last comma
378         if (sb.charAt(sb.length() - 1) == ',')
379             sb.setLength(sb.length() - 1);
380 
381         sb.append("); ").append(this.foundAnnotations.toString());
382         return sb.toString();
383     }

406     protected Class toClass(Type t) {
407         if (t == null)
408             return null;
409         else if (t instanceof Class)
410             return (Class) t;
411         else if (t instanceof ParameterizedType)
412             return (Class) ((ParameterizedType) t).getRawType();
413         else if (t instanceof WildcardType)
414             return (Class) ((WildcardType) t).getUpperBounds()[0];
415 
416         throw new RuntimeException("Generic TypeVariable not supported!");
417 
418     }

491     public Class getConcreteType() {
492         Embedded e = getAnnotation(Embedded.class);
493         if (e != null) {
494             Class concrete = e.concreteClass();
495             if (concrete != Object.class) {
496                 return concrete;
497             }
498         }

502             Class concrete = p.concreteClass();
